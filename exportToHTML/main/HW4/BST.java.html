package main.HW4;

public class BST<E extends Comparable<E>> {
    // you'll want to define a Node type here
    // ... but you have to design it yourself!
    // Node defined as class BSTNode
    @SuppressWarnings("hiding")
    class BSTNode<E> {
        private E data;
        private BSTNode<E> left;
        private BSTNode<E> right;

        public BSTNode() { // goes to next constructor
            this(null);
        }

        public BSTNode(E val) { // goes to next constructor
            this(val, null, null);
        }

        public BSTNode(E val, BSTNode<E> lc, BSTNode<E> rc) { // final constructor
            data = val;
            left = lc;
            right = rc;
        }

        public E getData() { // if i feel like being fancy
            return data;
        }

        public void setData(E newData) { // ^^^
            data = newData;
        }

        public BSTNode<E> getlc() {
            return left;
        }

        public BSTNode<E> getrc() {
            return right;
        }

        public boolean haslc() {
            return left != null;
        }

        public boolean hasrc() {
            return right != null;
        }

        public boolean isLeaf() {
            return (left == null) && (right == null);
        }

        public int getHeight() {
            return height(this);
        }

        private int height(BSTNode<E> node) {
            int resHeight = 0;
            if (node != null) {
                resHeight = 1 + Math.max(height(node.left), height(node.right));
            }
            return resHeight;
        }

        public int size() {
            if (this.data == null) {
                return 0;
            } else if (this.getlc() == null && this.getrc() == null) {
                return 1;
            } else if (left != null && right != null) {
                return 1 + left.size() + right.size();
            } else if (right == null) {
                return 1 + left.size();
            } else {
                return 1 + right.size();
            }
        }
    }

    private BSTNode<E> root;

    public BST() {
        root = null;
    }

    public BST(E elt) {
        root = new BSTNode<E>(elt);
    }

    public int size() {
        if (root == null) {
            return 0;
        }
        return root.size();
    }

    public int height() {
        if (root == null) {
            return 0;
        }
        return root.getHeight();
    }

    public boolean isPerfect() {
        if (root == null) {
            return true;
        }
        return perfHelper(root);
    }

    public boolean perfHelper(BSTNode<E> node) {
        if (node.isLeaf()) {
            return true;
        } else if (node.getlc() != null && node.getrc() != null && node.left.getHeight() == node.right.getHeight()) {
            return perfHelper(node.left) && perfHelper(node.right);
        } else {
            return false;
        }
    }

    public boolean isDegenerate() {
        if (root == null) {
            return true;
        }
        return degenHelper(root);
    }

    public boolean degenHelper(BSTNode<E> node) {
        if (node.isLeaf()) {
            return true;
        } else if (node.haslc() && node.hasrc()) {
            return false;
        } else if (node.haslc()) {
            return degenHelper(node.left);
        } else if (node.hasrc()) {
            return degenHelper(node.right);
        } else {
            return false;
        }
    }

    public E insert(E elt) {
        if (elt == null) {
            return null;
        }
        if (root == null) {
            root = new BSTNode<E>(elt);
            return null;
        } else {
            BSTNode<E> parent = null;
            BSTNode<E> current = root;
            while (current != null) {
                if (elt.compareTo(current.data) < 0) {
                    parent = current;
                    current = current.left;
                } else if (elt.compareTo(current.data) > 0) {
                    parent = current;
                    current = current.right;
                } else {
                    return current.data;
                }
            }
            if (elt.compareTo(parent.data) < 0) {
                parent.left = new BSTNode<E>(elt);
            } else if (elt.compareTo(parent.data) > 0) {
                parent.right = new BSTNode<E>(elt);
            } else if (parent.data.equals(elt)) {
                return parent.data;
            }
        }
        return null;
    }

    public E lookup(E elt) throws RuntimeException {
        if (root == null || !this.contains(elt)) {
            throw new RuntimeException("No such elt");
        } else {
            BSTNode<E> current = root;
            while (current != null) {
                if (elt.compareTo(current.data) < 0) {
                    current = current.left;
                } else if (elt.compareTo(current.data) > 0) {
                    current = current.right;
                } else {
                    return current.data;
                }
            }
            throw new RuntimeException("No such elt");
        }
    }

    public boolean contains(E elt) {
        if (root == null) {
            return false;
        }
        return contHelper(elt, root);
    }

    public boolean contHelper(E elt, BSTNode<E> node) {
        if (node == null) {
            return false;
        } else if (node.haslc() && elt.compareTo(node.data) < 0) {
            return contHelper(elt, node.left) || false;
        } else if (node.hasrc() && elt.compareTo(node.data) > 0) {
            return contHelper(elt, node.right) || false;
        } else if (node.data == elt) {
            return true;
        }
        return false;
    }

    public E remove(E elt) {
        if (elt == null) {
            return null;
        }
        if (root == null || !this.contains(elt)) {
            return null;
        } else if (root.size() == 1) {
            BSTNode<E> old = root;
            root = null;
            return old.data;
        } else if (root.data == elt) {
            BSTNode<E> old = root;
            root = root.right;
            findLeaf(root, 0).left = old.left;
            return old.data;
        }
        return removeHelper(root, elt, null);
    }

    private E removeHelper(BSTNode<E> node, E tgt, BSTNode<E> parent) {
        if (node.data == tgt) {
            seek(node, parent);
            return node.data;
        } else if (node.data.compareTo(tgt) < 0) {
            return removeHelper(node.right, tgt, node);
        } else if (node.data.compareTo(tgt) > 0) {
            return removeHelper(node.left, tgt, node);
        }
        return null;
    }

    private void seek(BSTNode<E> old, BSTNode<E> parent) {
        int direction;
        if (parent.left == old) {
            direction = 0;
        } else {
            direction = 1;
        }
        if (old.haslc() && old.hasrc()) {
            if (direction == 0) {
                parent.left = old.right;

                findLeaf(old.right, 0).left = old.left;
            } else {
                parent.right = old.left;
                findLeaf(old.left, 1).right = old.right;
            }
        } else if (old.isLeaf()) {
            if (parent.left == old) {
                parent.left = null;
            } else {
                parent.right = null;
            }
        } else {
            if (old.haslc() && direction == 0) {
                parent.left = old.left;
            } else if (old.hasrc() && direction == 0) {
                parent.left = old.right;
            } else if (old.haslc() && direction == 0) {
                parent.right = old.left;
            } else if (old.hasrc() && direction == 1) {
                parent.right = old.right;
            }
        }
    }

    private BSTNode<E> findLeaf(BSTNode<E> node, int direction) {
        if (direction == 0) {
            while (node.left != null) {
                node = node.left;
            }
            return node;
        } else {
            while (node.right != null) {
                node = node.right;
            }
            return node;
        }
    }

    public static BST<Integer> makeDegenerate(int size) {
        BST<Integer> Degen = new BST<Integer>();
        for (int i = 0; i < size; i++) {
            Degen.insert(i);
        }
        return Degen;
    }

    public static BST<Integer> makePerfect(int height) {
        BST<Integer> tree = new BST<Integer>();

        if (height == 0) {
            return tree;
        } else if (height==1) {
            tree.insert(1);
            return tree;
        }

        int treesize = (int) (Math.pow(2, height)-1);
        int center = (treesize+1)/2;
        int deviation = center/2;

        tree.insert(center);
        makePerfHelp(center, deviation, tree);
        return tree;
    }

    public static void makePerfHelp(int center, int deviation, BST<Integer> tree){
        int lc = center-deviation;
        int rc = center+deviation;
        if (deviation == 1) {
            tree.insert(lc);
            tree.insert(rc);
        } else {
            tree.insert(lc);
            tree.insert(rc);

            makePerfHelp(lc, deviation/2, tree);
            makePerfHelp(rc, deviation/2, tree);
        }
    }
